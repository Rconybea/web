template<typename Unit>
concept unit_concept
#include <unit_concept.hpp>

concept for a Unit type, suitable for use with the quantity template

Example: 

using namespace xo::unit;
static_assert(unit_concept<units::day>);

template<typename Unit>
concept basis_unit_concept
#include <unit_concept.hpp>

concept for a Unit type, that contains exactly one basis dimension

Example: 

using namespace xo::unit
static_assert(basis_unit_concept<units::volatility_250d>);
